How to use TEA

The Transparency Exchange API (TEA) is an open standard for discovering SBOMs. You can query this server programmatically using libtea, the Python TEA client: 

# pip install libtea
from libtea import TeaClient

with TeaClient.from_well_known("pypi.sbomify.com") as client:
    # Search for a package by PURL
    results = client.search_product_releases(
        "PURL", "pkg:pypi/maturin@1.8.1"
    )

    for release in results.results:
        # Get the SBOM collection for this release
        collection = client.get_product_release_collection_latest(
            release.uuid
        )
        for artifact in collection.artifacts:
            print(f"{artifact.name} ({artifact.type})")

            # Download the SBOM with checksum verification
            fmt = artifact.formats[0]
            client.download_artifact(
                fmt.url, f"./{artifact.name}",
                verify_checksums=fmt.checksums,
            )

Or use the CLI: tea-cli inspect "urn:tei:purl:pypi.sbomify.com:pkg:pypi/maturin@1.8.1"
